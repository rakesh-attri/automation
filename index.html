3 SF Helpers — Login & Global Setup ▶
utils/sf-helpers.jsjs
import { chromium } from '@playwright/test';
import fs from 'fs';
import 'dotenv/config';

/**
 * Global Setup: Login to Salesforce and save auth state.
 * Playwright reuses this across all tests (no re-login).
 */
export default async function globalSetup() {
  const browser = await chromium.launch({ headless: true });
  const page = await browser.newPage();

  console.log('🔐 Logging into Salesforce...');

  await page.goto(`${process.env.SF_URL}/lightning/page/home`);
  await page.fill('#username', process.env.SF_USERNAME);
  await page.fill('#password', process.env.SF_PASSWORD);
  await page.click('#Login');

  // Wait for Lightning to fully load
  await page.waitForSelector('[data-id="AppNavigation"]', { timeout: 30000 });

  // Save auth state — reused by all tests
  await page.context().storageState({ path: './reports/.auth-state.json' });
  await browser.close();

  console.log('✅ Salesforce login successful. Auth state saved.');
}

/** Helper: Wait for SF Lightning spinner to disappear */
export async function waitForSFLoad(page) {
  await page.waitForFunction(() =>
    !document.querySelector('.forceListViewManagerSpinner') &&
    !document.querySelector('.slds-spinner')
  , { timeout: 15000 }).catch(() => {}); // Non-fatal
}

/** Helper: Navigate to SF app via App Launcher */
export async function navigateToApp(page, appName) {
  await page.click('[title="App Launcher"]');
  await page.fill('[placeholder="Search apps and items..."]', appName);
  await page.click(`text=${appName}`);
  await waitForSFLoad(page);
}

1 orchestrator.js — Master Coordinator ▶
agents/orchestrator.jsjs
import fs from 'fs/promises';
import path from 'path';
import { PlannerAgent } from './planner-agent.js';
import { GeneratorAgent } from './generator-agent.js';
import { HealerAgent } from './healer-agent.js';
import { runTests, parseTestResults } from '../utils/reporter-utils.js';

export class Orchestrator {
  constructor(taskFilePath) {
    this.taskFile = taskFilePath;
    this.planner = new PlannerAgent();
    this.generator = new GeneratorAgent();
    this.healer = new HealerAgent();
    this.maxHealAttempts = 3;
  }

  async run() {
    console.log(`\n🚀 Orchestrator: Starting pipeline for ${this.taskFile}\n`);

    // STEP 1: Read task file
    const taskContent = await fs.readFile(this.taskFile, 'utf8');
    const taskName = extractTaskName(taskContent);

    // STEP 2: Planner explores SF and produces a plan
    console.log('🔍 Step 1/4: Planner exploring Salesforce...');
    const planFile = await this.planner.plan(taskContent, taskName);

    // STEP 3: Generator reads plan, writes .spec.js
    console.log('⚡ Step 2/4: Generator writing Playwright tests...');
    const specFile = await this.generator.generate(planFile, taskName);

    // STEP 4: Run tests
    console.log('▶  Step 3/4: Running tests...');
    let results = await runTests(specFile);

    // STEP 5: Heal if needed
    let healAttempt = 0;
    while (results.failed > 0 && healAttempt < this.maxHealAttempts) {
      healAttempt++;
      console.log(`🩹 Step 4/4: Healer fixing failures (attempt ${healAttempt})...`);
      await this.healer.heal(specFile, results.errors);
      results = await runTests(specFile);
    }

    // FINAL SUMMARY
    console.log(`\n✅ Pipeline Complete!`);
    console.log(`   Passed: ${results.passed} | Failed: ${results.failed}`);
    console.log(`   Spec:   ${specFile}`);
    console.log(`   Report: npm run report:allure\n`);

    return results;
  }
}
2 planner-agent.js — MCP-Powered SF Explorer ▶
agents/planner-agent.jsjs
import Anthropic from '@anthropic-ai/sdk';
import fs from 'fs/promises';
import path from 'path';
import 'dotenv/config';

export class PlannerAgent {
  constructor() {
    this.client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
  }

  async plan(taskContent, taskName) {
    // Load memory files for context
    const memory = await fs.readFile('./memory/sf-selectors.md', 'utf8');
    const frameworkCtx = await fs.readFile('./memory/framework-memory.md', 'utf8');

    const prompt = `
You are a Playwright test PLANNING agent for Salesforce Lightning.
You have access to MCP browser tools to explore the live SF org.

## Framework Memory
${frameworkCtx}

## Known SF Selectors Library
${memory}

## Task to Plan
${taskContent}

## Your Job
1. Use browser tools to navigate to the relevant SF Lightning page
2. Take screenshots and analyze the DOM
3. Identify all form fields, buttons, toasts, and list views
4. Write a DETAILED Markdown test plan with:
   - Step-by-step actions
   - Exact locators (prefer: role, aria-label, placeholder, text)
   - Assertions to verify at each step
   - Expected success/failure states
   - Screenshot capture points

Output ONLY the Markdown plan. Start with # Test Plan: [task name]
`;

    const response = await this.client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 8192,
      messages: [{ role: 'user', content: prompt }],
    });

    const plan = response.content[0].text;
    const planFile = `./specs/${taskName}-plan.md`;
    await fs.writeFile(planFile, plan, 'utf8');

    console.log(`   📄 Plan saved: ${planFile}`);
    return planFile;
  }
}
3 generator-agent.js — Test Code Writer ▶
agents/generator-agent.jsjs
import Anthropic from '@anthropic-ai/sdk';
import fs from 'fs/promises';
import 'dotenv/config';

export class GeneratorAgent {
  constructor() {
    this.client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
  }

  async generate(planFile, taskName) {
    const plan = await fs.readFile(planFile, 'utf8');

    const prompt = `
You are a Playwright JS test GENERATOR for Salesforce Lightning.

## Input Plan
${plan}

## Requirements — Generate tests that:
1. Use Playwright's @playwright/test framework
2. Import allure-js-commons for rich reporting
3. Use ONLY semantic/accessible locators:
   - page.getByRole('button', { name: 'New' })
   - page.getByLabel('Opportunity Name')
   - page.getByPlaceholder('Search...')
   - page.getByText('Success') / page.locator('[aria-label="..."]')
4. Include ALLURE annotations:
   - allure.epic(), allure.feature(), allure.story()
   - allure.step() for each major action
   - allure.attachment() for screenshots
5. Screenshot on every failure using test.info().attach()
6. Proper beforeAll/afterAll hooks for SF login (use storageState)
7. Detailed expect() assertions at every step
8. Add waitForSFLoad() after navigation actions
9. Add retry logic for flaky SF elements

## Output
Generate ONLY executable Playwright JS code. 
No markdown, no explanations.
File starts with: import { test, expect } from '@playwright/test';
`;

    const response = await this.client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 8192,
      messages: [{ role: 'user', content: prompt }],
    });

    let testCode = response.content[0].text;
    // Strip markdown fences if present
    testCode = testCode.replace(/```javascript|```js|```/g, '').trim();

    const specFile = `./tests/${taskName}.spec.js`;
    await fs.writeFile(specFile, testCode, 'utf8');
    console.log(`   🧪 Test generated: ${specFile}`);
    return specFile;
  }
}
4 healer-agent.js — Auto-Fix Failing Tests ▶
agents/healer-agent.jsjs
import Anthropic from '@anthropic-ai/sdk';
import fs from 'fs/promises';
import 'dotenv/config';

export class HealerAgent {
  constructor() {
    this.client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
  }

  async heal(specFile, errors) {
    const testCode = await fs.readFile(specFile, 'utf8');
    const sfSelectors = await fs.readFile('./memory/sf-selectors.md', 'utf8');

    const prompt = `
You are a Playwright TEST HEALER agent.
A test is failing. Your job is to fix the broken locators or logic.

## Failing Test Code
${testCode}

## Error Messages
${errors.join('\n\n')}

## Known SF Selectors Library
${sfSelectors}

## Rules for Healing
1. ONLY fix what's broken — don't rewrite working parts
2. Prefer more stable locators: aria-label > text > css
3. Add .first() or .nth(0) if multiple elements match
4. Add explicit waits before flaky elements: waitFor({ state: 'visible' })
5. For toast messages: waitForSelector('.toastMessage') then getText
6. For dropdowns: use selectOption() or click + getByRole('option')
7. Comment each fix with: // HEALED: reason

Return ONLY the fixed test code. No markdown fences.
`;

    const response = await this.client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 8192,
      messages: [{ role: 'user', content: prompt }],
    });

    let healed = response.content[0].text.replace(/```(js|javascript)?/g, '').trim();
    await fs.writeFile(specFile, healed, 'utf8');
    console.log(`   🩹 Test healed and saved: ${specFile}`);
  }
}

2 framework-memory.md — Persistent Agent Context ▶
memory/framework-memory.mdmarkdown
# Framework Memory — SF Agentic Automation
_Updated by agents automatically. Do not delete._

## Framework Rules (Always Follow)
1. Use ONLY semantic locators — never CSS classes or XPath with IDs
2. Always call waitForSFLoad() after navigating or clicking
3. Salesforce Lightning spinners must disappear before interacting
4. Toast messages auto-dismiss — assert within 3 seconds of action
5. All tests must be reusable across any SF org (no org-specific IDs)
6. Auth state is stored in ./reports/.auth-state.json

## Salesforce Lightning Patterns

### Modal/Dialog Handling
- New record modal: role=dialog
- Close: getByRole('button', { name: 'Cancel' })
- Save: getByRole('button', { name: 'Save' })

### Toast Message Pattern
```javascript
await expect(page.locator('.toastMessage')).toBeVisible();
await expect(page.locator('.toastMessage')).toContainText('was created');
```

### Navigation Pattern
```javascript
await page.getByRole('link', { name: 'Leads' }).click();
await waitForSFLoad(page);
```

### Dropdown / Picklist Pattern
```javascript
await page.getByLabel('Lead Status').click();
await page.getByRole('option', { name: 'Open - Not Contacted' }).click();
```

## Known Agent Outputs
| Task File | Plan File | Spec File | Last Run |
|-----------|-----------|-----------|----------|
| tasks/lead-creation.md | specs/lead-creation-plan.md | tests/lead-creation.spec.js | - |

## Lessons Learned
- SF custom components sometimes need force click — use click({ force: true })
- List views default to "Recently Viewed" — switch to "All [Object]" first
- Inline edit requires double-click on field value

utils/reporter-utils.jsjs
import { test } from '@playwright/test';
import * as allure from 'allure-js-commons';
import { execSync } from 'child_process';

/**
 * Attach a screenshot to test and Allure report
 */
export async function captureScreenshot(page, name = 'screenshot') {
  const screenshot = await page.screenshot({ fullPage: true });
  // Attach to Playwright HTML report
  await test.info().attach(name, {
    body: screenshot,
    contentType: 'image/png',
  });
  // Attach to Allure report
  await allure.attachment(name, screenshot, 'image/png');
}

/**
 * Wrap actions with Allure step tracking
 */
export async function step(name, fn) {
  return allure.step(name, fn);
}

/**
 * Run Playwright tests and parse results
 */
export async function runTests(specFile) {
  try {
    const cmd = `npx playwright test ${specFile} --reporter=json`;
    const output = execSync(cmd, { encoding: 'utf8' });
    const results = JSON.parse(output);
    return {
      passed: results.stats.expected,
      failed: results.stats.unexpected,
      errors: extractErrors(results),
    };
  } catch (e) {
    return { passed: 0, failed: 1, errors: [e.message] };
  }
}

function extractErrors(results) {
  return results.suites
    ?.flatMap(s => s.specs)
    ?.flatMap(spec => spec.tests)
    ?.filter(t => t.status === 'unexpected')
    ?.map(t => t.results[0]?.error?.message || 'Unknown error') || [];
}

import { test, expect } from '@playwright/test';
import * as allure from 'allure-js-commons';
import { captureScreenshot, step } from '../utils/reporter-utils.js';
import { waitForSFLoad } from '../utils/sf-helpers.js';

test.use({ storageState: './reports/.auth-state.json' });

test.describe('Lead Creation Flow', () => {

  test.beforeEach(async ({ page }) => {
    await allure.epic('CRM');
    await allure.feature('Lead Management');
    await allure.story('Create New Lead');
    await allure.severity('critical');
  });

  test('should create a new Lead and verify in list view', async ({ page }) => {
    const timestamp = Date.now();
    const leadName = `Agentic Test-${timestamp}`;

    // STEP 1: Navigate to Leads
    await step('Navigate to Leads tab', async () => {
      await page.goto(process.env.SF_URL);
      await page.getByRole('link', { name: 'Leads' }).click();
      await waitForSFLoad(page);
      await captureScreenshot(page, 'leads-list-view');
    });

    // STEP 2: Click New
    await step('Click New button', async () => {
      await page.getByRole('button', { name: 'New' }).click();
      await page.getByRole('dialog').waitFor({ state: 'visible' });
    });

    // STEP 3: Fill the form
    await step('Fill Lead creation form', async () => {
      await page.getByLabel('First Name').fill('Agentic');
      await page.getByLabel('Last Name').fill(`Test-${timestamp}`);
      await page.getByLabel('Company').fill('SF Agentic Framework Inc.');
      await page.getByLabel('Email').fill('agentic.test@sf-framework.com');
      await page.getByLabel('Phone').fill('+91-9876543210');
      await captureScreenshot(page, 'lead-form-filled');
    });

    // STEP 4: Save
    await step('Save the Lead record', async () => {
      await page.getByRole('button', { name: 'Save' }).click();

      // Assert success toast
      const toast = page.locator('.toastMessage');
      await expect(toast).toBeVisible({ timeout: 10000 });
      await expect(toast).toContainText('Lead');
      await captureScreenshot(page, 'lead-saved-toast');
    });

    // STEP 5: Verify detail page
    await step('Verify Lead detail page', async () => {
      await waitForSFLoad(page);
      await expect(page.getByRole('heading', { name: leadName })).toBeVisible();
      await captureScreenshot(page, 'lead-detail-page');
    });

    // STEP 6: Verify in list view
    await step('Verify Lead in list view', async () => {
      await page.getByRole('link', { name: 'Leads' }).click();
      await waitForSFLoad(page);
      await expect(page.getByRole('link', { name: leadName })).toBeVisible();
      await captureScreenshot(page, 'lead-in-list-view');
    });
  });
});

#!/usr/bin/env node
import { Orchestrator } from '../agents/orchestrator.js';
import 'dotenv/config';

// Parse --task argument
const args = process.argv.slice(2);
const taskIdx = args.indexOf('--task');
const taskFile = taskIdx !== -1
  ? args[taskIdx + 1]
  : 'tasks/lead-creation.md';  // Default task

console.log(`
╔══════════════════════════════════════════╗
║   SF Agentic Automation Framework        ║
║   Powered by Claude + Playwright MCP     ║
╚══════════════════════════════════════════╝

Task: ${taskFile}
`);

const orchestrator = new Orchestrator(taskFile);
orchestrator.run()
  .then(results => {
    process.exit(results.failed > 0 ? 1 : 0);
  })
  .catch(err => {
    console.error('❌ Fatal error:', err);
    process.exit(1);
  });
